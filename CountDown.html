function calculateYearsAndDaysLeft() { let futureDate = new Date('2077-10-19'); let currentDate = new Date(); let diff = futureDate.getTime() - currentDate.getTime(); let years = Math.floor(diff / (1000 * 60 * 60 * 24 * 365.25)); let days = Math.floor(diff / (1000 * 60 * 60 * 24)) % 365; console.log("Years left: " + years); console.log("Days left: " + days); } calculateYearsAndDaysLeft();
